305      public void goToPosition(double targetXPosition, double targetYPosition, double robotPower, double desiredRobotOrientation, double allowableDistanceError ){
306          targetXPosition *= COUNTS_PER_INCH;
307          targetYPosition *= COUNTS_PER_INCH;
308          allowableDistanceError *= COUNTS_PER_INCH;
309          double blPower = 0; // motor speed
310          double brPower = 0; // motor speed
311          double flPower = 0; // motor speed
312          double frPower = 0; // motor speed
313          double pivotCorrectionAdj = .1; // constant to scale down pivot correction angle to work with setting powers for mecanum drive motors
314          double distanceToXTarget = targetXPosition - globalPositionUpdate.returnXCoordinate();
315          double distanceToYTarget = targetYPosition - globalPositionUpdate.returnYCoordinate();
316          double distance = Math.hypot(distanceToXTarget, distanceToYTarget);
317          while (opModeIsActive() && distance > allowableDistanceError ) { //correct heading too
318              distance = Math.hypot(distanceToXTarget, distanceToYTarget);
319              distanceToXTarget = targetXPosition - globalPositionUpdate.returnXCoordinate();
320              distanceToYTarget = targetYPosition - globalPositionUpdate.returnYCoordinate();
321              double robotMovementAngle = Math.toDegrees(Math.atan2(distanceToXTarget, distanceToYTarget));
322              double robotMovmentXComponent = calculateX(robotMovementAngle - globalPositionUpdate.returnOrientation(), robotPower);
323              double robotMovmentYComponent = calculateY(robotMovementAngle - globalPositionUpdate.returnOrientation(), robotPower);
324              double pivotCorrection = (desiredRobotOrientation - globalPositionUpdate.returnOrientation())*pivotCorrectionAdj;
325              //double[] powers = {robotMovmentYComponent, robotMovmentYComponent, robotMovmentYComponent, robotMovmentYComponent};//array for powers
326              blPower = robotMovmentYComponent - robotMovmentXComponent + pivotCorrection;
327              flPower = robotMovmentYComponent + robotMovmentXComponent + pivotCorrection;
328              brPower = robotMovmentYComponent + robotMovmentXComponent - pivotCorrection;
329              frPower = robotMovmentYComponent - robotMovmentXComponent - pivotCorrection;
330              //set powers to motors to move
331              double maxMotorPower = Math.max(Math.max(Math.max(Math.abs(flPower), Math.abs(frPower)), Math.abs(blPower)), Math.abs(brPower));
332  
333              if (Math.abs(maxMotorPower) > 1) {
334                  flPower = (flPower / maxMotorPower)*robotPower;
335                  frPower = (frPower / maxMotorPower) *robotPower;
336                  blPower = (blPower / maxMotorPower) *robotPower;
337                  brPower = (brPower / maxMotorPower)*robotPower;
338              } else if(Math.abs(maxMotorPower) < .03) {
339                  flPower = 0;
340                  frPower = 0;
341                  blPower = 0;
342                  brPower = 0;
343              }
344              flMotor.setPower(flPower);
345              frMotor.setPower(frPower);
346              blMotor.setPower(blPower);
347              brMotor.setPower(brPower);
348              telemetry.addData("X Position", globalPositionUpdate.returnXCoordinate() / COUNTS_PER_INCH);
349              telemetry.addData("Y Position", globalPositionUpdate.returnYCoordinate() / COUNTS_PER_INCH);
350              telemetry.addData("Orientation (Degrees)", globalPositionUpdate.returnOrientation());
351              telemetry.addData("XComponent: ", robotMovmentXComponent/.9);
352              telemetry.addData("YComponent: ", robotMovmentYComponent/.9);
353              telemetry.addData("vertical right", globalPositionUpdate.verticalRightEncoderWheelPosition);
354              telemetry.addData("vertical left", globalPositionUpdate.verticalLeftEncoderWheelPosition);
355              telemetry.addData("horizontal", globalPositionUpdate.normalEncoderWheelPosition);
356              telemetry.addData("Pivot Correction: ", pivotCorrection);
357              telemetry.update();
358          }
359      }

417      /**
418       * Calculate the power in the x direction
419       * @param desiredAngle angle on the x axis
420       * @param speed robot's speed
421       * @return the x vector
422       */
423      private double calculateX(double desiredAngle, double speed) {
424          return Math.sin(Math.toRadians(desiredAngle)) * speed;
425      }
426  
427      /**
428       * Calculate the power in the y direction
429       * @param desiredAngle angle on the y axis
430       * @param speed robot's speed
431       * @return the y vector
432       */
433      private double calculateY(double desiredAngle, double speed) {
434          return Math.cos(Math.toRadians(desiredAngle)) * speed;
435      }

445      public void hingeWobbleGoalArm(int num){
446          if (num < 0)// -1 want to Hinge (bring down arm)
447          {
448              wobbleArmHingeL.setPower(-1);
449              wobbleArmHingeR.setPower(1);
450              sleep(1500);
451              wobbleArmHingeL.setPower(0);
452              wobbleArmHingeR.setPower(0);
453          }
454          if (num > 0) {// +1 want to unHinge (bring arm back up)
455              wobbleArmHingeL.setPower(-1); // open arm
456              wobbleArmHingeR.setPower(1); // open arm
457              sleep(1500); // continue for a second
458              wobbleArmHingeL.setPower(0); // stop servo
459              wobbleArmHingeR.setPower(0); // open arm
460          }
461      }
462      public void gripWobbleGoal(int num){
463          if (num > 0)// +1 to Grip
464          {
465              wobbleArmGripL.setPower(1); // open arm
466              wobbleArmGripR.setPower(-1); // open arm
467              sleep(1000); // continue for a second
468              wobbleArmGripL.setPower(0); // stop servo
469              wobbleArmGripR.setPower(0); // open arm
470          }
471          if (num < 0) {// -1 to ungrip
472              wobbleArmGripL.setPower(-1); // open arm
473              wobbleArmGripR.setPower(1); // open arm
474              sleep(1000); // continue for a second
475              wobbleArmGripL.setPower(0); // stop servo
476              wobbleArmGripR.setPower(0); // open arm
477          }
478      }
